'use strict';

/**
 * @module Authentication
*/

Object.assign(exports, {
  authenticate: { func: authenticate, authentication: false, type: 'post' },
  refreshToken: { func: refreshToken, authentication: true, type: 'get' }
});

/**
 * @function POST /api/authenticate
 * @description FB user registration and login
 *
 * @param {String} email
 * @param {String} token
 * @param {String} type - The type of token (ie. 'Facebook')
 * @param {String} [name] - Required for first time user
 * @param {String} [avatarUrl]
 * @return {Object.<Boolean, String, String, Number>} An object containing a flag representating whether or not a user exists where true means user already exists, an access token, refresh token, and uid
 */
async function authenticate({ services, context }, params) {
  const res = await services.auth.authenticate(context, params);
  return res;
}

/**
 * @function POST /api/refreshToken
 * @description creates a new access token
 * @summary Authentication is required
 *
 * @param {String} refreshToken
 *
 * @return {String} Should return updated access token
*/
async function refreshToken({ services, context }, params) {
  const accessToken = await services.auth.refreshToken(context, params);
  return { accessToken };
}
