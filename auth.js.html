'use strict';

/**
 * @module Authentication
*/

Object.assign(exports, {
  authenticateFb: { func: authenticateFb, authentication: false, type: 'post' },
  refreshToken: { func: refreshToken, authentication: false, type: 'get' }
});

/**
 * @function POST /api/authenticateFb
 * @description FB user registration and login
 *
 * @param {String} uid - Should be facebook id
 * @param {String} [name] - Required for first time user
 * @param {String} [email] - Required for first time user
 * @param {Number} [birthdate] - Required for first time user and should be a ts
 * @param {String} token
 * @param {String} type - The type of token (ie. 'Facebook')
 * @param {String} [.gender] - Should only be either "M" or "F" case-sensitive
 * @param {String} [.avatarUrl]
 * @param {String} [.description]
 * @return {Object.<Boolean, String, String>} An object containing a flag representating whether or not a user exists where true means user already exists, an access token, and a refresh token
 */
async function authenticateFb({ services, context }, params) {
  const res = await services.auth.authenticateFb(context, params);
  return res;
}

/**
 * @function POST /api/refreshToken
 * @description creates a new access token
 * 
 * @param {String} uid 
 * @param {String} refreshToken
 * @return {String} Should return updated access token
*/
async function refreshToken({ services, context }, params) {
  const accessToken = await services.auth.refreshToken(context, params);
  return accessToken;
}
