'use strict';

/**
 * @module Comment
*/

Object.assign(exports, {
  getComments: { func: getComments, authentication: false, type: 'get' },
  createComment: { func: createComment, authentication: true, type: 'post' },
  deleteComment: { func: deleteComment, authentication: true, type: 'delete' }
});

/**
 * Get 10 comments from an event
 * @param {Context}
 * @param {Object} params
 * @param {Number} params.eid - The event id
 * @param {Number} params.offset - The comment id to begin returning from
 * @return {Object.<Comment[]>} comments - An array of Comments
 */
async function getComments({ services, context }, params) {
  const comments = await services.comment.get(context, params);
  return { comments };
}

/**
 * Create new comment
 * @param {Context}
 * @param {Object} params
 * @param {Number} params.eid
 * @param {String} params.body
 * @return {Object.<Number>} id - The id of the created comment
 */
async function createComment({ services, context }, params) {
  const id = await services.comment.create(context, params);
  return { id };
}

/**
 * Delete a comment in an Event
 * @param {Context}
 * @param {Object} params
 * @param {number} params.id - The comment id
 * @return {Object.<Void>}
 */
function deleteComment({ services, context }, params) {
  return services.comment.delete(context, params);
}
