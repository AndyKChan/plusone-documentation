'use strict';

/**
 * @module Comment
 */

Object.assign(exports, {
  getComments: { func: getComments, authentication: false, type: 'get' },
  createComment: { func: createComment, authentication: true, type: 'post' },
  deleteComment: { func: deleteComment, authentication: true, type: 'delete' }
});

/**
 * 
 * @function POST /api/getComments
 *
 * @description Get 10 comments from an event
 *
 * @param {Number} eid - The event id
 * @param {Number} offset - The comment id to begin returning from
 * @return {Comment[]} comments - An array of Comments
 */
async function getComments({ services, context }, params) {
  const comments = await services.comment.get(context, params);
  return { comments };
}

/**
 *
 * @function POST /api/createComment
 * @description Creates a new Comment
 * @summary Authentication is required
 *
 * @param {Number} eid
 * @param {String} body
 * @return {Number} id - The id of the created comment
 */
async function createComment({ services, context }, params) {
  const id = await services.comment.create(context, params);
  return { id };
}

/**
 *
 * @function POST /api/deleteComment
 * @description Delete a comment in an Event
 * @summary Authentication is required
 *
 * @param {number} id - The comment id
 * @return {Void}
 */
function deleteComment({ services, context }, params) {
  return services.comment.delete(context, params);
}
