'use strict';

/**
 * @module Event
*/

Object.assign(exports, {
  createEvent,
  getEvent,
  updateEvent,
  deleteEvent,
  getUserRSVPs,
  getEventRSVPs,
  updateRSVP,
  createRSVP
});

/**
 * Create a new event
 * @param {Context} context
 * @param {Object} params
 * @param {String} params.sport
 * @param {Number} params.ts - Timestamp
 * @param {Number} params.lat
 * @param {Number} params.lon
 * @param {String} params.title
 * @param {String} [params.description]
 * @param {String} [params.difficulty] - Should be one of beginner, intermediate, or expert
 * @param {String} [params.url] - The URL of the event's picture
 * @return {Object.<Number>} id - The id of the created event
 */
async function createEvent({ services, context }, params) {
  const id = await services.event.create(context, params);
  return { id };
}

/**
 * Get an event
 * @param {Context} context
 * @param {Object} params
 * @param {Number} params.eid - The event id
 * @return {Object.<Event>} event
 */
async function getEvent({ services, context }, params) {
  const event = await services.event.get(context, params);
  return { event: event.toPlainObject() };
}

/**
 * Update an event
 * @param {Context} context
 * @param {Object} params
 * @param {Number} params.eid
 * @param {String} [params.sport]
 * @param {Number} [params.ts]
 * @param {Number} [params.lat]
 * @param {Number} [params.lon]
 * @param {String} [params.title]
 * @param {String} [params.description]
 * @param {String} [params.difficulty]
 * @param {String} [params.url]
 * @return {Object.<Void>}
 */
function updateEvent({ services, context }, params) {
  return services.event.update(context, params);
}

/**
 * Delete an event
 * @param {Context} context
 * @param {Object} params
 * @param {Number} params.eid
 * @return {Object.<Void>}
 */
async function deleteEvent({ services, context }, params) {
  await services.event.delete(context, params);
}

/**
 * Get a list of RSVPs for a User
 * @param {Context} context
 * @param {Object} params
 * @param {Number} params.uid - The id of the User
 * @return {Object.<RSVP[]>} rsvps - A list of RSVPs
 */
async function getUserRSVPs({ services, context }, params) {
  const rsvps = await services.event.getUserRSVPs(context, params);
  return { rsvps };
}

/**
 * Get a list of RSVPs for an Event
 * @param {Context} context
 * @param {Object} params
 * @param {Number} params.eid - The id of the event
 * @return {Object.<RSVP[]>} - A list of RSVPs
 */
async function getEventRSVPs({ services, context }, params) {
  const rsvps = await services.event.getEventRSVPs(context, params);
  return { rsvps };
}

/**
 * Update an RSVP response
 * @param {Context} context
 * @param {Object} params
 * @param {Number} params.uid - The uid of the rsvp owner
 * @param {Number} params.eid - The id of the event 
 * @param {String} params.status - The updated attendance status of user
 * @param {String} [params.reason] - Reason will be null if status changes
 * @return {Object.<Void>}
*/
function updateRSVP({ services, context }, params) {
  return services.event.updateRSVP(context, params);
}

/**
 * Post RSVP response to Event
 * @param {Context} context
 * @param {Object} params
 * @param {integer} params.eid The id of the event
 * @param {String} params.status The attendence status of user
 * @param {String} [params.reason] The reason for stated status
 * @return {Object.<Void>}
*/
async function createRSVP({ services, context }, params) {
  await services.event.createRSVP(context, params);
}
