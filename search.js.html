'use strict';

const Aigle = require('aigle');

Object.assign(exports, {
  getEvents: { func: getEvents, authentication: false, type: 'get' },
  getEventsByLocation: { func: getEventsByLocation, authentication: false, type: 'get' }
});

/**
 * @function POST /api/getEvents
 * @description Queries for events and can be filtered by location, sport, difficulty, and date
 *
 * @param {Number} [lat]
 * @param {Number} [lon]
 * @param {String} [sport]
 * @param {String} [difficulty]
 * @param {String} [date]
 * @return {Object.<Array.<Event>>} An object named "events" that contains an array of event objects
 */
async function getEvents({ context, services }, params) {
  const res = await services.search.getEvents(context, params);
  const events = await Aigle.map(res.hits.hits, hit => {
    return hit._source;
  });
  return { events }; 
}

/**
 * @function POST /api/getEventsByLocation
 * @description Queries for events in a specific location and can be filtered by sport, difficulty, and date
 *
 * @param {Number} lat_t Top left lat
 * @param {Number} lon_t Top left lon
 * @param {Number} lat_b Bottom right lat
 * @param {Number} lon_b Bottom right lon
 * @param {String} [sport]
 * @param {String} [difficulty]
 * @param {String} [date]
 * @return {Object.<Array.<Event>>} An object named "events" that contains an array of event objects
 */
async function getEventsByLocation({ context, services }, params) {
  const res = await services.search.getEventsByLocation(context, params);
  const events = await Aigle.map(res.hits.hits, hit => {
    return hit._source;
  });
  return { events }; 
}
