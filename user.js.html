'use strict';

/**
 * @module User
 */

/**
 * If authentication is required, a header must be included in the form of:
 * ex. Cookie:uid=40
 */

Object.assign(exports, {
  getUserWithId: { func: getUserWithId, authentication: false, type: 'get' },
  getUserWithUid: { func: getUserWithUid, authentication: false, type: 'get' },
  updateUser: { func: updateUser, authentication: true, type: 'put' },
  blockUser: { func: blockUser, authentication: true, type: 'post' },
  unblockUser: { func: unblockUser, authentication: true, type: 'delete' },
  isBlockedByEid: { func: isBlockedByEid, authentication: false, type: 'get' }
});

/**
 * Get a User with id
 *
 * @description POST /api/getUserWithID
 * 
 * @param {Number} id
 * @return {User} User
*/
async function getUserWithId({ services, context }, params) {
  const user = await services.user.getWithId(context, params);
  return { user: user.toPlainObject() };
}

/**
 * Get a User with uid
 * 
 * @description POST /api/getUserWithUid
 *
 * @param {Number} uid
 * @return User {User}
*/
async function getUserWithUid({ services, context }, params) {
  const user = await services.user.getWithUid(context, params);
  return { user: user.toPlainObject() };
}

/**
 * Update a User with uid
 * 
 * @description POST /api/updateUser
 * @summary Authentication is required
 *
 * @param {String} [name]
 * @param {String} [email]
 * @param {Number} [birthdate]
 * @param {String} [gender]
 * @param {String} [avatarUrl]
 * @param {String} [description]
 * @return {Void}
*/
function updateUser({ services, context }, params) {
  return services.user.update(context, params);
}

/**
 * Block a User
 * 
 * @description POST /api/blockUser
 * @summary Authentication is required
 *
 * @param {Number} blockeeId - The id of the target user to be blocked
 * @return {Void}
*/
function blockUser({ services, context }, params) {
  return services.user.block(context, params);
}

/**
 * Unblock a User
 * 
 * @description POST /api/unblockUser
 * @summary Authentication is required
 *
 * @param {Number} blockeeId - The id of the target user to be unblocked
 * @return {Void}
*/
function unblockUser({ services, context }, params) {
  return services.user.unblock(context, params);
}

/**
 * Check if User is blocked
 * 
 * @description POST /api/isBlockedByEid
 * 
 * @param {Number} id - The id of the event
 * @param {Number} uid - The uid of potential blockee
 * @return {Boolean} isBlocked
 */
async function isBlockedByEid({ services, context }, params) {
  const isBlocked = await services.user.isBlockedByEid(context, params);
  return { isBlocked };
}
