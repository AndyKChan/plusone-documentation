'use strict';

/**
 * @module User
*/

Object.assign(exports, {
  getUserWithId,
  getUserWithUid,
  updateUser,
  blockUser,
  unblockUser,
  isBlockedByEid
});

/**
 * Get a User with id
 * @param {Context}
 * @param {Object} params
 * @param {Number} params.id
 * @return {Object.<User>} User
*/
async function getUserWithId({ services, context }, params) {
  const user = await services.user.getWithId(context, params);
  return { user: user.toPlainObject() };
}

/**
 * Get a User with uid
 * @param {Context}
 * @param {Object} params
 * @param {Number} params.uid
 * @return {Object.<User>} User
*/
async function getUserWithUid({ services, context }, params) {
  const user = await services.user.getWithUid(context, params);
  return { user: user.toPlainObject() };
}

/**
 * Update a User with uid
 * @param {Context}
 * @param {Object} params
 * @param {String} [params.name]
 * @param {String} [params.email]
 * @param {Number} [params.birthdate]
 * @param {String} [params.gender]
 * @param {String} [params.avatarUrl]
 * @param {String} [params.description]
 * @return {Object.<Void>}
*/
function updateUser({ services, context }, params) {
  return services.user.update(context, params);
}

/**
 * Block a User
 * @param {Context}
 * @param {Object} params
 * @param {Number} params.blockeeId - The id of the target user to be blocked
 * @return {Object.<Void>}
*/
function blockUser({ services, context }, params) {
  return services.user.block(context, params);
}

/**
 * Unblock a User
 * @param {Context}
 * @param {Object} params
 * @param {Number} params.blockeeId - The id of the target user to be unblocked
 * @return {Object.<Void>}
*/
function unblockUser({ services, context }, params) {
  return services.user.unblock(context, params);
}

/**
 * Check if User is blocked
 * @param {Context}
 * @param {Object} params
 * @param {Number} params.id - The id of the event
 * @return {Object.<Boolean>} isBlocked - True if blocked, false otherwise
 */
async function isBlockedByEid({ services, context }, params) {
  const isBlocked = await services.user.isBlockedByEid(context, params);
  return { isBlocked };
}
